#! /bin/bash

for(( i = 1; i <= 12; i++ ))
do
	if [ $1 -eq 1 ] || [ $1 -eq 3 ] || [ $1 -eq 5 ] || [ $1 -eq 7 ] || [ $1 -eq 10 ] || [ $1 -eq 12 ] then
		for(( j = 1;j <= 31;j++))
		do
		mkdir -p /e/$1/$i/$j
		done
	elif [ $1 -eq 4 ] || [ $1 -eq 6 ] || [ $1 -eq 9 ] || [ $1 -eq 11 ] then
		for(( j = 1;j <= 30;j++ ))
		do
		mkdir -p /e/$1/$i/$j
		done
	else
		for(( j = 1;j <= 28;j++ ))
		do
		mkdir -p /e/$1/$i/$j
		done
	fi
done
